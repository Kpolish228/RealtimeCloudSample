/* Simple Web usage example */
import { Client } from "realtime-cloud";

const client = new Client({ key: "YOUR_API_KEY" });

const call = await client.call.create({
  roomId: "demo-room",
  video: true,
  audio: true,
});

call.join(document.getElementById("app"));
        

// Swift iOS example
import UIKit
import RealtimeCloudSDK

let client = RCClient(config: RCConfig(apiKey: "YOUR_API_KEY"))
